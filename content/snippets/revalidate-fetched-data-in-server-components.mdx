## Static

```jsx

const getData = () => {
    const res = await fetch("https://api/data")
    return res.json() // getStaticProps (fetches data during build)
}
```

## SSR

```jsx

const getData = () => {
    const res = await fetch("https://api/data", {
        cache: 'no-store' //getServerSideProps (data fresh on every request)
    })
    return res.json()
}
```

## ISR

```jsx

const getData = () => {
    const res = await fetch("https://api/data", {
        next: {revalidate: 10} // getServerSideProps (cache data refreshes every 10s)
    })
    return res.json()
}
```
