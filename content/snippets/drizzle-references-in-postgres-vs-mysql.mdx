---
title: Drizzle references in PostgreSQL vs. MySQL
publishedAt: "2023-08-25"
summary: Differences between the definition of references between schemas in PostgreSQL and MySQL.
---

# PostgreSQL (Neon)

```jsx
import { int, mysqlTable, serial, text } from "drizzle-orm/mysql-core"

export const users = mysqlTable("users", {
  id: serial("id").primaryKey(),
})

export const posts = mysqlTable("posts", {
  id: serial("id").primaryKey(),
  text: text("text"),
  user_id: int("user_id").references(() => users.id),
})
```

# MySQL (PlanetScale)

```jsx
import { relations } from "drizzle-orm"
import { int, mysqlTable, serial, text } from "drizzle-orm/mysql-core"

export const users = mysqlTable("users", {
  id: serial("id").primaryKey(),
})

export const posts = mysqlTable("posts", {
  id: serial("id").primaryKey(),
  text: text("text"),
  user_id: int("user_id"),
})

export const usersRelations = relations(users, ({ many }) => ({
  posts: many(posts),
}))

export const postsRelations = relations(posts, ({ one }) => ({
  user: one(users, {
    fields: [posts.user_id],
    references: [users.id],
  }),
}))
```

This will allow you to go from this:

```jsx
const user = await db.select().from(users).where(eq(users.id, 1))
```

To this:

```jsx
const user = await db.query.users.findFirst({
  with: {
    posts: true,
  },
})
```
